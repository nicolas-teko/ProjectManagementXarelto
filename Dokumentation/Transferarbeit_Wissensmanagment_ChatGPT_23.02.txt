using Microsoft.EntityFrameworkCore;
using ProjectManagementXarelto.Core.Models;

namespace ProjectManagementXarelto.Data.Context;

public class AppDbContext : DbContext
{
    public DbSet<User> Users => Set<User>();
    public DbSet<Project> Projects => Set<Project>();
    public DbSet<Information> Informations => Set<Information>();
    public DbSet<Comment> Comments => Set<Comment>();
    public DbSet<InfoTag> InfoTags => Set<InfoTag>();
    public DbSet<InfoLink> InfoLinks => Set<InfoLink>();

    public AppDbContext(DbContextOptions<AppDbContext> options)
        : base(options)
    {
    }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);

        // ===== USER =====
        modelBuilder.Entity<User>()
            .Property(u => u.DisplayName)
            .IsRequired()
            .HasMaxLength(200);

        // ===== PROJECT =====
        modelBuilder.Entity<Project>()
            .Property(p => p.Name)
            .IsRequired()
            .HasMaxLength(200);

        modelBuilder.Entity<Project>()
            .Property(p => p.CustomerName)
            .IsRequired()
            .HasMaxLength(200);

        modelBuilder.Entity<Project>()
            .HasOne(p => p.ProjectLeadUser)
            .WithMany()
            .HasForeignKey(p => p.ProjectLeadUserId)
            .OnDelete(DeleteBehavior.Restrict);

        modelBuilder.Entity<Project>()
            .HasMany(p => p.Informations)
            .WithOne(i => i.Project!)
            .HasForeignKey(i => i.ProjectId)
            .OnDelete(DeleteBehavior.Cascade);

        // ===== INFORMATION =====
        modelBuilder.Entity<Information>()
            .Property(i => i.Text)
            .IsRequired();

        modelBuilder.Entity<Information>()
            .HasMany(i => i.Comments)
            .WithOne(c => c.Information!)
            .HasForeignKey(c => c.InformationId)
            .OnDelete(DeleteBehavior.Cascade);

        modelBuilder.Entity<Information>()
            .HasMany(i => i.Tags)
            .WithOne(t => t.Information!)
            .HasForeignKey(t => t.InformationId)
            .OnDelete(DeleteBehavior.Cascade);

        modelBuilder.Entity<Information>()
            .HasMany(i => i.Links)
            .WithOne(l => l.Information!)
            .HasForeignKey(l => l.InformationId)
            .OnDelete(DeleteBehavior.Cascade);

        // ===== INFOTAG =====
        modelBuilder.Entity<InfoTag>()
            .Property(t => t.Tag)
            .IsRequired()
            .HasMaxLength(100);

        modelBuilder.Entity<InfoTag>()
            .HasIndex(t => new { t.InformationId, t.Tag })
            .IsUnique(); // kein doppeltes Tag pro Info

        // ===== INFOLINK =====
        modelBuilder.Entity<InfoLink>()
            .Property(l => l.Url)
            .IsRequired();
    }
}

# ADR 0001 – Architektur, Datenhaltung und MVP-Umsetzung (WinForms + EF Core + SQLite)

**Status:** Accepted  
**Datum:** 2026-02-15  
**Projekt:** ProjectManagementXarelto

## Kontext

Es soll eine sehr einfache, lokal laufende Windows-Desktop-Applikation entstehen, um Projekte anzulegen und projektbezogene Informationselemente inkl. Kommentaren zu verwalten. Rollenbasierte Rechte steuern, wer Projekte/Infos/Kommentare erstellen darf. Overengineering soll vermieden werden, dennoch soll die Struktur sauber und wartbar bleiben.

## Entscheidungen

### 1) Frontend-Technologie
**Entscheidung:** WinForms (.NET 8)  
**Begründung:** Schnellstes, simplest UI-Framework für Windows-only Desktop, gutes Designer-Tooling in Visual Studio 2022.

### 2) Projektstruktur (Solution-Aufteilung)
**Entscheidung:** 3 Projekte in einer Solution
- `ProjectManagementXarelto.App` (WinForms UI)
- `ProjectManagementXarelto.Core` (Models + Services/Businessregeln)
- `ProjectManagementXarelto.Data` (EF Core, DbContext, Migrations, Seed)

**Begründung:** Saubere Trennung von UI, Domänenlogik und Persistenz; bleibt dennoch leichtgewichtig.

### 3) Datenbank
**Entscheidung:** SQLite + EF Core (.NET 8 kompatibel)  
**Begründung:** Lokal, filebasiert, kein Server/Setup nötig, ideal für MVP.

### 4) EF Core Migrations & Design-Time Setup
**Entscheidung:** Migrations werden per Package Manager Console erstellt/ausgeführt (`Add-Migration`, `Update-Database`).  
Zusätzlich wird eine `IDesignTimeDbContextFactory<AppDbContext>` verwendet.

**Begründung:** EF Core benötigt zur Design-Time (Migration-Erstellung) eine Möglichkeit, `DbContextOptions` zu instanziieren.

### 5) Authentifizierung (MVP)
**Entscheidung:** Kein echtes Login. Zwei hardcoded Benutzer (Projektleiter/Projektmanager) werden in die DB geseedet; Login erfolgt später über UI per Rollen-Button.  
**Begründung:** Minimale Komplexität, entspricht Aufgabenstellung/MVP-Ziel.

### 6) Rollen & Berechtigungen (MVP-Regeln)
**Entscheidung:**
- Projekt erstellen: nur `ProjectLead`
- Information erstellen: `ProjectLead` oder `ProjectManager`
- Information bearbeiten: `ProjectLead` oder `ProjectManager`
- Kommentar erstellen: `ProjectLead` oder `ProjectManager`
- Kommentare sind nicht editierbar
- Projekte sind nach Erstellung nicht editierbar

**Begründung:** Entspricht den fachlichen Anforderungen.

### 7) Audit-Felder / Änderbarkeit
**Entscheidung:**
- `Project`: `CreatedAt`, `CreatedByUserId`; **kein** `UpdatedAt/UpdatedBy` (Projekt readonly nach Erstellung)
- `Information`: `CreatedAt`, `CreatedByUserId`, plus `UpdatedAt`, `UpdatedByUserId` (einzige editierbare Entität)
- `Comment`: `CreatedAt`, `CreatedByUserId`; **kein** Update (Kommentare readonly)

**Begründung:** Klare Nachvollziehbarkeit, minimaler Aufwand, exakt wie gefordert.

### 8) Datenmodell für Tags
**Entscheidung:** Tags werden als separate Entität `InfoTag` modelliert (pro Information mehrere Zeilen), kein globales Tag-Lexikon.  
Constraint: Unique Index `(InformationId, Tag)` (kein doppeltes Tag pro Info).  
Businessregel: **max. 3 Tags pro Information** wird in der Applikationslogik validiert (nicht DB-seitig).

**Begründung:** Einfach, gut filterbar pro Projekt/Information, kein String-Splitting, keine globale Tag-Verwaltung nötig.

### 9) Datenmodell für URLs (Bild/Dokument)
**Entscheidung:** URLs werden normalisiert als `InfoLink` gespeichert:
- `Type` (Enum: Image/Document)
- `Url` (String)
- 1..n Links pro Information

**Begründung:** Sauber, erweiterbar, kein CSV/JSON im Feld.

### 10) Delete-Verhalten (Referenzintegrität)
**Entscheidung (EF Fluent API):**
- `Project -> Informations`: Cascade
- `Information -> Comments/Tags/Links`: Cascade
- `Project -> ProjectLeadUser`: Restrict

**Begründung:** Beim Löschen einer Information sollen ihre abhängigen Datensätze (Kommentare/Tags/Links) mitgelöscht werden. Projektleiter darf nicht gelöscht werden, solange Projekte referenzieren. (MVP UI bietet keine Delete-Funktion; Verhalten ist für technische Konsistenz definiert.)

### 11) SQLite DB-Datei & Pfad
**Entscheidung:** Aktuell wird eine relative Connection (`Data Source=projectmanagement.db`) verwendet; dadurch liegt die DB im Working Directory (typisch `bin/Debug/net8.0-windows/`).  
**Begründung:** Minimaler Setup für MVP.  
**Hinweis:** Für stabilen Betrieb kann später auf einen fixen Pfad (z.B. AppData) umgestellt werden.

### 12) Git / GitHub
**Entscheidung:** Repository wird aus Visual Studio heraus erstellt (Default VisualStudio .gitignore). Zusätzlich soll `*.db` in `.gitignore` aufgenommen werden.  
**Begründung:** Verhindert das versehentliche Committen der SQLite-Datei; bin/obj sind ohnehin ignoriert.

- EF Core Migrations sind reproduzierbar; Design-Time Factory verhindert Migration-Fehler.
- Rollen-Login ist bewusst vereinfacht, kann später ersetzt/erweitert werden.
- Tag- und URL-Modellierung ist normalisiert und unterstützt Filter/Suche ohne Sonderlogik.

using System.Linq;
using Microsoft.EntityFrameworkCore;
using ProjectManagementXarelto.Core.Models;

namespace ProjectManagementXarelto.App
{
    public partial class ProjectDetailsForm : Form
    {
        private readonly int _projectId; // Erklärung: Id des Projekts, das angezeigt werden soll

        public ProjectDetailsForm(int projectId)
        {
            InitializeComponent();
            _projectId = projectId;                     // Erklärung: Projekt-Id speichern

            this.Load += ProjectDetailsForm_Load;       // Erklärung: Beim Laden des Forms Daten aus DB holen
        }

        private void ProjectDetailsForm_Load(object? sender, EventArgs e)
        {
            // Erklärung: Projekt inkl. Projektleiter und Informationen aus der Datenbank laden
            var project = Program.DbContext.Projects
                .Include(p => p.ProjectLeadUser)
                .Include(p => p.Informations)
                .FirstOrDefault(p => p.Id == _projectId);

            if (project == null)
            {
                MessageBox.Show("Projekt wurde nicht gefunden.");
                Close();
                return;
            }

            // Erklärung: Kopfbereich mit Projektinformationen füllen
            lblProjectName.Text = $"Projekt: {project.Name}";
            lblCustomerName.Text = $"Kunde: {project.CustomerName}";
            lblProjectLead.Text = $"Projektleiter: {project.ProjectLeadUser?.DisplayName ?? "-"}";

            txtInitialDescription.Text = project.InitialDescription; // Erklärung: Initialbeschreibung anzeigen (readonly)

            LoadInformations(project); // Erklärung: Liste der Informations-Elemente füllen
        }

        private void LoadInformations(Project project)
        {
            // Erklärung: Informationen des Projekts nach Erstellungsdatum sortieren
            var infos = project.Informations
                .OrderBy(i => i.CreatedAt)
                .ToList();

            lstInformations.Items.Clear(); // Erklärung: Liste leeren

            foreach (var info in infos)
            {
                // Erklärung: Einfache Anzeigezeile pro Information erzeugen
                var displayText = $"{info.CreatedAt:yyyy-MM-dd HH:mm} - {info.Text}";
                lstInformations.Items.Add(displayText);
            }

            // Erklärung: Falls keine Informationen vorhanden sind, Hinweis anzeigen
            if (!infos.Any())
            {
                lstInformations.Items.Add("(Keine Informationen vorhanden)");
            }
        }
    }
}

Im Konstruktor wird die Projekt-Id gespeichert.

Beim Laden (ProjectDetailsForm_Load) wird das Projekt inkl. Projektleiter und seiner Informations geladen.

Labels + TextBox werden gefüllt.

LoadInformations schreibt eine simple Liste der Informations-Elemente in die ListBox.

namespace ProjectManagementXarelto.App
{
    public partial class InformationEditForm : Form
    {
        private readonly int _informationId; // Erklärung: Id der Information, die bearbeitet wird
        private Information? _info;          // Erklärung: Referenz auf das geladene Info-Objekt

        public InformationEditForm(int informationId)
        {
            InitializeComponent();
            _informationId = informationId;             // Erklärung: Id speichern

            this.Load += InformationEditForm_Load;      // Erklärung: Beim Laden Daten holen
            btnSave.Click += BtnSave_Click;             // Erklärung: Klick auf "Speichern"
            btnCancel.Click += BtnCancel_Click;         // Erklärung: Klick auf "Abbrechen"
        }

        private void InformationEditForm_Load(object? sender, EventArgs e)
        {
            // Erklärung: Information aus der DB laden
            _info = Program.DbContext.Informations
                .FirstOrDefault(i => i.Id == _informationId);

            if (_info == null)
            {
                MessageBox.Show("Information wurde nicht gefunden.");
                DialogResult = DialogResult.Cancel;
                Close();
                return;
            }

            // Erklärung: Textfeld mit aktuellem Inhalt füllen
            txtInfoText.Text = _info.Text;

            // Erklärung: einfache Meta-Info anzeigen (Erstellt / Geändert)
            var meta = $"Erstellt: {_info.CreatedAt:yyyy-MM-dd HH:mm}";
            if (_info.UpdatedAt.HasValue)
            {
                meta += $" | Geändert: {_info.UpdatedAt:yyyy-MM-dd HH:mm}";
            }
            lblMeta.Text = meta;
        }

        private void BtnSave_Click(object? sender, EventArgs e)
        {
            if (_info == null)
                return;

            var user = SessionContext.CurrentUser; // Erklärung: aktuell eingeloggter Benutzer

            if (user == null)
            {
                MessageBox.Show("Kein Benutzer angemeldet.");
                return;
            }

            if (string.IsNullOrWhiteSpace(txtInfoText.Text))
            {
                MessageBox.Show("Der Informationstext darf nicht leer sein.");
                return;
            }

            // Erklärung: Änderungen an der Information übernehmen
            _info.Text = txtInfoText.Text.Trim();
            _info.UpdatedAt = DateTime.UtcNow;
            _info.UpdatedByUserId = user.Id;

            Program.DbContext.SaveChanges(); // Erklärung: Änderungen speichern

            DialogResult = DialogResult.OK;  // Erklärung: Erfolg anrufendem Form signalisieren
            Close();
        }

        private void BtnCancel_Click(object? sender, EventArgs e)
        {
            DialogResult = DialogResult.Cancel; // Erklärung: ohne Speichern schließen
            Close();
        }
    }
